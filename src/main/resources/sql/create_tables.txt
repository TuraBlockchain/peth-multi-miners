networks=CREATE TABLE APP.NETWORKS (ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), URL VARCHAR(256) NOT NULL)
accounts=CREATE TABLE APP.ACCOUNTS (ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), ADDRESS VARCHAR(256) NOT NULL, PASSPHRASE VARCHAR(2048) NOT NULL)
mpath=CREATE TABLE APP.MPATH (ID INTEGER PRIMARY KEY NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), ACCOUNT VARCHAR(256) NOT NULL, PATH VARCHAR(1024) NOT NULL)
